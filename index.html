#!/bin/bash # Detect the operating system detect_os() { OS="Unknown" if [[ "$OSTYPE" == "linux-gnu"* ]]; then OS="Linux" elif [[ "$OSTYPE" == "darwin"* ]]; then OS="MacOS" elif [[ "$OSTYPE" == "cygwin" || "$OSTYPE" == "msys" || "$OSTYPE" == "win32" ]]; then OS="Windows" else OS=$(uname -s) fi echo "$OS" } # Command to run on Linux/macOS bash_command() { echo "Running on Linux/macOS:" date } # Command to run on Windows (PowerShell) powershell_command() { echo "Running on Windows:" powershell.exe -Command "Get-Date" } # Main logic OS=$(detect_os) if [[ "$OS" == "Windows" ]]; then powershell_command else bash_command fi
